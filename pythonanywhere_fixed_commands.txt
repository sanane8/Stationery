# PythonAnywhere Fixed Commands
# Use these instead of the previous commands

# 1. Create virtual environment
mkvirtualenv --python=python3.11 stationery

# 2. Activate virtual environment
workon stationery

# 3. Go to project directory
cd stationery_pythonanywhere

# 4. Install dependencies
pip install -r requirements_pythonanywhere.txt

# 5. Run migrations with PythonAnywhere settings
python manage_pythonanywhere.py migrate

# 6. Collect static files
python manage_pythonanywhere.py collectstatic --noinput

# 7. Create superuser (this should work now!)
python manage_pythonanywhere.py createsuperuser

# 8. Test the application
python manage_pythonanywhere.py check

# 9. Generate secure SECRET_KEY for production
python -c "import secrets; print('SECRET_KEY=' + secrets.token_urlsafe(50))"
