# Download files directly to PythonAnywhere
# Run these commands in Bash console

cd stationery_pythonanywhere

# Download the PythonAnywhere-specific manage file
curl -o manage_pythonanywhere.py https://raw.githubusercontent.com/django/django/main/django/conf/project_template/manage.py

# Create a basic WSGI file
cat > pythonanywhere_wsgi.py << 'EOF'
import os
import sys
from pathlib import Path

# Add your project directory to Python path
project_home = '/home/yourusername/stationery_pythonanywhere'
if project_home not in sys.path:
    sys.path.insert(0, project_home)

# Set the Django settings module
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'stationery_tracker.settings_pythonanywhere')

# Configure Django
from django.core.wsgi import get_wsgi_application
application = get_wsgi_application()
EOF

# Create requirements file
cat > requirements_pythonanywhere.txt << 'EOF'
Django==5.1.6
gunicorn==23.0.0
whitenoise==6.10.0
psycopg2-binary==2.9.10
django-cors-headers==4.7.0
django-humanize==0.1.2
python-decouple==3.8
EOF

echo "Files created successfully!"
ls -la
